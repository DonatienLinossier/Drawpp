cmake_minimum_required(VERSION 3.10)

# Set project name
project(drawppCompiler C)

# Specify the path to the SDL2 installation
if(WIN32)
    set(CMAKE_PREFIX_PATH "${CMAKE_SOURCE_DIR}/libs/SDL2-2.30.8/x86_64-w64-mingw32")
endif ()

# Find SDL2
find_package(SDL2 REQUIRED)

# Include directories
include_directories(${SDL2_INCLUDE_DIRS})
include_directories(${CMAKE_SOURCE_DIR}/include)

# Add executable
add_executable(${PROJECT_NAME} src/main.c src/cursor.c src/sdlEncapsulation.c
)

# Link SDL2 libraries
target_link_libraries(${PROJECT_NAME} ${SDL2_LIBRARIES})
